@Entity
@Table (name="tab")
public class entityy {
	@Id
	public int id;
	
	@Column (name="name")
	public String name;
	
	@Column(name="address")
	public String address; 
}

@RestController
public class controll {
	
	@Autowired
	private reposit obj;
	
	@RequestMapping("/hell")
	public String gethell() {
		return "hello";
	}
	
	@GetMapping("/get")
	public List<entityy> get(){
		return obj.findAll();
	}
	
	@PostMapping("/post")
	public String post(@RequestBody entityy ob) {
		obj.save(ob);
		return "done";
	}
     
	@PutMapping("/put/{id}")
	public String put(@RequestBody entityy os,@PathVariable int id) {
		entityy put = obj.findById(id).orElse(null);
		if (put!=null) {
			put.setId(os.getId());
			put.setName(os.getName());
			put.setAddress(os.getAddress());
			obj.save(put);
			return"ok";
			
			}
		else {
			
			return "notnull";
		}
	}
	
	@DeleteMapping("/del/{id}")
	public String delete (@PathVariable int id) {
		entityy delete = obj.findById(id).orElse(null);
		if (delete!=null) {
			obj.delete(delete);
			return "success";
		}
		else {
			return "notnull";
		}
		
	}
}
